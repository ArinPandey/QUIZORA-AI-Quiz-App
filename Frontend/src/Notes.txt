javascriptexport const quizQuestions = [

🎮 src/components/Quiz/QuizApp.jsx (Main Brain)
Purpose: Central state management and game flow control
🏗️ State Variables:
javascriptconst [gameState, setGameState] = useState('start');     // Controls which screen shows
const [currentQuestion, setCurrentQuestion] = useState(0); // Current question index
const [selectedAnswer, setSelectedAnswer] = useState(null); // User's current selection
const [answers, setAnswers] = useState([]);              // All user answers
const [score, setScore] = useState(0);                   // Current score
const [questions, setQuestions] = useState([]);          // Loaded questions
const [showAnswer, setShowAnswer] = useState(false);     // Show correct answer?
🔧 Key Functions:
1. startQuiz()
javascriptconst startQuiz = () => {
  setGameState('quiz');      // Switch to quiz screen
  setCurrentQuestion(0);     // Start from question 1
  setSelectedAnswer(null);   // Clear any selection
  setAnswers([]);           // Clear previous answers
  setScore(0);              // Reset score
  setShowAnswer(false);     // Hide answer feedback
};
When called: User clicks "Start Quiz"
What it does: Resets all state and begins the quiz
2. handleAnswerSelect(answerIndex)
javascriptconst handleAnswerSelect = (answerIndex) => {
  if (showAnswer) return;        // Prevent changes after submission
  setSelectedAnswer(answerIndex); // Store user's choice (0-3)
};
When called: User clicks an answer option
Parameters: answerIndex - which option they clicked (0-3)
What it does: Records user selection, prevents changes after submission
3. submitAnswer()
javascriptconst submitAnswer = () => {
  if (selectedAnswer === null) return; // Must select something
  
  const isCorrect = selectedAnswer === questions[currentQuestion].correctAnswer;
  
  // Create detailed answer record
  const newAnswer = {
    questionId: questions[currentQuestion].id,
    question: questions[currentQuestion].question,
    selectedAnswer,              // What user chose
    correctAnswer: questions[currentQuestion].correctAnswer,
    options: questions[currentQuestion].options,
    isCorrect,                  // Boolean: right or wrong
    category: questions[currentQuestion].category
  };

  setAnswers([...answers, newAnswer]); // Add to answers array
  if (isCorrect) setScore(score + 1);  // Increment score if correct
  setShowAnswer(true);                 // Show answer feedback
};
When called: User clicks "Submit Answer"
What it does:

Checks if answer is correct
Creates detailed answer record
Updates score if correct
Shows visual feedback

4. nextQuestion()
javascriptconst nextQuestion = () => {
  if (currentQuestion < questions.length - 1) {
    // More questions available
    setCurrentQuestion(currentQuestion + 1);
    setSelectedAnswer(null);
    setShowAnswer(false);
  } else {
    // Quiz complete
    setGameState('results');
  }
};
When called: User clicks "Next Question" or "View Results"
What it does: Advances to next question or shows results
5. previousQuestion()
javascriptconst previousQuestion = () => {
  if (currentQuestion > 0) {
    // Remove last answer from records
    const lastAnswer = answers[answers.length - 1];
    if (lastAnswer && lastAnswer.isCorrect) {
      setScore(score - 1);  // Subtract from score if it was correct
    }
    setAnswers(answers.slice(0, -1)); // Remove last answer
    setCurrentQuestion(currentQuestion - 1);
    setSelectedAnswer(null);
    setShowAnswer(false);
  }
};
When called: User clicks "Previous"
What it does: Goes back one question, removes last answer from records
6. restartQuiz()
javascriptconst restartQuiz = () => {
  setGameState('start');     // Back to start screen
  // Reset all state to initial values
  setCurrentQuestion(0);
  setSelectedAnswer(null);
  setAnswers([]);
  setScore(0);
  setShowAnswer(false);
};
When called: User clicks "Take Quiz Again"
What it does: Complete reset to initial state
🎭 Screen Rendering Logic:
javascriptswitch (gameState) {
  case 'start':   return <StartScreen onStart={startQuiz} ... />;
  case 'quiz':    return <QuizScreen question={...} ... />;
  case 'results': return <ResultsScreen score={...} ... />;
}
